% \larrow{fillColor}
\newcommand{\larrow}[1]
{\begin{tikzpicture}[scale=0.58]
     \filldraw[fill=#1!100,draw=#1!100!black] (0,0) -- (0.2,0) -- (0.3,0.2) -- (0.2,0.4) -- (0,0.4) -- (0.1,0.2) -- cycle;
 \end{tikzpicture}
}

% \mycircle{fillColor}
\newcommand{\mycircle}[1]
{\textcolor{#1}{\textbullet}
}

% \mybullet{fillColor}
\newcommand{\mybullet}[1]
{ 
    \raisebox{0.25\height}[0pt][0pt] { \mycircle{#1} }
}

% \lightform{fillColor}
\newcommand{\lightform}[1]
{\begin{tikzpicture}[scale=0.008]
     \filldraw[fill=#1!100,draw=#1!100!black] (0,0) -- (30,0) -- (30.0,26.5) -- (22.5,7.5) -- (7.5,7.5) -- (7.5,22.5) -- (26.5,30.0) -- (0,30) -- cycle;
 \end{tikzpicture}
}


%----------------------------------------------------------------------------------------
% Custom sections
%----------------------------------------------------------------------------------------

% \begin{noPageBreakBox} <stuff that won't be broken across pages> \end{noPageBreakBox}
\newenvironment{noPageBreakBox}
{
    \par\noindent\begin{minipage}{\linewidth}
}
{
    \end{minipage}\par
}

% \begin{cvSection}{Title} \end{cvSection}
\newenvironment{cvSection}[1]
{
    \vspace{2pt}
    {\large\textbf{{\uppercase{#1}}}}
    \vspace{6pt}
}
{
    \vspace{12pt}
}

% \begin{cvEvent}[underlineColor]{Title, Company, Years} \end{cvEvent}
\newenvironment{cvEvent}[4][softColor]
{
    \vspace{2pt}
    \textcolor{black}{\textbf{#2}} \hfill \textcolor{black}{#3}, \textcolor{backgroundColor}{#4}
    \vspace{5pt}
    \textcolor{#1}{\hrule}
    \vspace{5pt}
}
{
    \vspace{9pt}
}

% \begin{cvEventNoYears}[underlineColor]{Title, Company} \end{cvEventNoYears}
\newenvironment{cvEventNoYears}[3][softColor]
{
    \vspace{2pt}
    \textcolor{black}{\textbf{#2}} \hfill \textcolor{black}{#3}
    \vspace{5pt}
    \textcolor{#1}{\hrule}
    \vspace{5pt}
}
{
    \vspace{9pt}
}

% \cvEventBullet{Sentence}
\newcommand{\cvEventBullet}[1]
{
\nopagebreak
\begin{itemize} \item[\mybullet{softColor}] #1 \end{itemize}
}

\newcommand{\cvEventNoBullet}[1]
{
\nopagebreak
\vspace{0.5mm}
#1
\vspace{0.5mm}
}

% \cvEventBulletSub{Sentence}
\newcommand{\cvEventBulletSub}[1]
{
\begin{itemize} \item[] \begin{itemize} \item[\mybullet{softerColor}] #1 \end{itemize} \end{itemize}
}

% \begin{sidebarSection}[underlineColor=white]{Title} \end{sidebarSection}
\newenvironment{sidebarSection}[2][white]
{
    \begin{center}
    \setulcolor{#1}
    \setul{7pt}{.4pt}% 7pt to make this roughly match the hrule from \cvEventHeader
    \textcolor{black}{\large{\uppercase{\ul{#2}}}}\\[8pt]
}
{
    \end{center}
    \vspace{12pt}
}

%----------------------------------------------------------------------------------------
% Icon + text helpers
%----------------------------------------------------------------------------------------

\newcommand{\icon}[3] % icon helper
{
    \textcolor{#3}{\csname #1\endcsname}
}

\newcommand{\icontext}[4] % icon next to text
{
    \centering{\icon{#1}{#2}{#4}} \textcolor{black}{#3}
}

\newcommand{\iconhref}[5] % icon next text with link
{
    \centering{\icon{#1}{#2}{#5}} \href{#4}{\textcolor{black}{#3}}
}
